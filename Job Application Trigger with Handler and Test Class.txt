    public static void recordCountUpdate(Job_Application__c job)
    {	
         	 if(job.CandidateNumber__c!=null)

                                                from Job_Application__c group by CandidateNumber__c
                                                having CandidateNumber__c=:job.CandidateNumber__c ];

                can.Total_No_Of_JobApplications__c=(Integer)results[0].get('totalCount');
                update can;
            }
    }

}
