Apex Class

public class LightningApexClass 
{
    @AuraEnabled //this annotation make this function available to Aura or LWC Javascript
    public static string addAccount(String accName,String accPhone)
    {
        Account a=new Account();
        a.Name=accName;
        a.Phone=accPhone;
        insert a;
        System.debug(accName);
        return 'Account Created Successfuly with Id:'+a.Id;
    }

}



Aura CMP
<aura:component   controller="LightningApexClass"   implements="force:appHostable,flexipage:availableForAllPageTypes,flexipage:availableForRecordHome,force:hasRecordId,forceCommunity:availableForAllPageTypes,force:lightningQuickAction" access="global" >
	
    <aura:attribute name="accountName" type="String"></aura:attribute>
     <aura:attribute name="accountPhone" type="String"></aura:attribute>
    
    <lightning:card title="New Account">
        <div>
        	<lightning:input label="Enter Account Name"
                             value="{!v.accountName}">                
            </lightning:input>
            
            <lightning:input label="Enter Account Phone"
                             value="{!v.accountPhone}">
            </lightning:input>
            
            <lightning:button label="Save" title="Save" onclick="{!c.saveRecord}">
            </lightning:button>
        </div>
    
    </lightning:card>
</aura:component>



JavaScript
({
	saveRecord : function(component, event, helper)
    {
        //get values from Aura HTML to Javascript
        var name=component.get('v.accountName');      
        var phone=component.get('v.accountPhone');
        
        var action=component.get('c.addAccount');
        
        action.setParams({
            accName:name,
            accPhone:phone
        });
        
        action.setCallback(this,function(response){
            var result=response.getReturnValue();
            alert(result);
        });
        
        $A.enqueueAction(action);        
        
        
	}
})
