    public static void func14()
    {
        List<AggregateResult>  results=[select StageName,
                                        count(Id) RecordCount,
                                        sum(Amount) TotalAmount 
                                        from Opportunity 
                                        Group by 
                                        StageName
                                        having StageName!='Closed Lost' and sum(Amount)>500000];
        
        System.debug(results);
        for(AggregateResult r:results)
        {
            String stage=(String)r.get('StageName');
            Integer totalCount=(Integer)r.get('RecordCount');
            Decimal sumAmount=(Decimal)r.get('TotalAmount');
            System.debug('Stage Name:'+stage);
            System.debug('Record Count:'+totalCount);
            System.debug('Total Amount:'+sumAmount);
            System.debug('');
        }
        
    }
